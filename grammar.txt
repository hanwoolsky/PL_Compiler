prog ->	word ( ) block 
decls -> decls decl | 系
decl -> vtype word ; 
vtype -> int | char | 系
block -> { decls slist } | 系
slist -> slist stat | stat 
stat -> IF cond THEN block ELSE block | word = expr ; | EXIT expr ; | 系
cond -> expr > expr 
expr -> expr + fact | fact 	
fact ->	num | word 
word -> ([a-z]|[A-Z])* 
num	-> [0-9]*
